# Set up
1: Cd to project
2: run command: yarn
3: run command: yarn add -D babel-loader
4: run command: yarn prod   # Compile frontend resources

# Start first node	// http://localhost:3001
yarn start

# Start second node	// http://localhost:3002
yarn start2

# Start third node	// http://localhost:3003
yarn start3
